# Set dhcp server on private LAN's gateway router.
# Fill in the correct values from the worksheet for items enclosed in  .
# 2016-10-26
#-----------------------------------------------------------

set service dhcp-server disabled 'false'

set service dhcp-server shared-network-name LAN authoritative 'enable'

set service dhcp-server shared-network-name LAN subnet 'subnet-ID/CIDR-bits'  default-router 'router's-IP-address'
set service dhcp-server shared-network-name LAN subnet 'subnet-ID/CIDR-bits'  dns-server 'router's-IP-address'
set service dhcp-server shared-network-name LAN subnet 'subnet-ID/CIDR-bits'  domain-name 'INSIDE'
set service dhcp-server shared-network-name LAN subnet 'subnet-ID/CIDR-bits'  lease '86400'

set service dhcp-server shared-network-name LAN subnet 'subnet-ID/CIDR-bits'  start 'first-client-address'  stop 'last-client-address'
