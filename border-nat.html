# Set NAT on border router.
# If necessary, change "10.141.0.1" to the upstream router.
# Change "172.16.119" to the appropriate subnet.
# 2016-10-22
#-----------------------------------------------------------

# Hide LAN hosts behind router's IP address
set nat source rule 100 outbound-interface 'eth0'
set nat source rule 100 source address '10.0.141.0/8'
set nat source rule 100 translation address masquerade

set nat source rule 200 outbound-interface 'eth0'
set nat source rule 200 source address '172.16.0.0/16'
set nat source rule 200 translation address masquerade
